// 필요한 만큼 메모리를 확보해서 그 값을 1부터 시작하는 값으로 채워라.
// 메모리의 값을 모두 더한 후 화면에 출력하라.

int main()
//--------
{
	//save("main.cpp");

	while (true) {
		cout << " int 몇 개가 필요한가요? ";
		unsigned num;		// 음수 방지
		cin >> num;

		// 시스템에 필요한 메모리를 요청한다.
		// 성공 - 시작번지를 알려준다.
		// 실패 - 자원이 고갈되었다는 예외를 던진다.
		int* p;
		p = new int[num];

		for (int i = 0; i < num; ++i) {
			*(p+i) = i + 1;					// p[i]의 의미는 ==> *(p + sizeof(int)*i)
											// p[i] <-- 는 실제로는 존재하지 않는다. 사용자의 편의를 위해 존재하는 syntatic sugar
		}
	}
}

// [문제] 몇 개의 int값을 저장할 메모리가 필요한지 사용자에게 물어본다.
// 필요한 만큼 메모리를 확보해서 그 값을 1부터 시작하는 값으로 채워라.
// 메모리의 값을 모두 더한 후 화면에 출력하라.

//--------
int main()
//--------
{
	//save("main.cpp");

	while (true) {
		cout << " int 몇 개가 필요한가요? ";
		unsigned num;
		cin >> num;

		// 시스템에 필요한 메모리를 요청한다.
		// 성공 - 시작번지를 알려준다.
		// 실패 - 자원이 고갈되었다는 예외를 던진다.
		int* p;
		p = new int[num];

		for (int i = 0; i < num; ++i) {
			*(p+i) = i + 1;
		}

		int sum{};

		// for ( int num : p )	// <--- 왜 p에 빨간줄이 쳐지는가? p는 범위가 아니기 때문
		for (int i = 0; i < num; ++i) {
			sum += p[i];
		}

		cout << "1 부터 " << num << " 까지 합계 = " << sum << std::endl;
	}
}

int 정렬(const void* x, const void* y) {
	string s1 = *(string*)x;
	string s2 = *(string*)y;

	if (s1 < s2)
		return -1;
	if (s1 > s2)
		return 1;
	return 0;
}

// [문제] 2차원점 100개 생성
// 생성 위치 x, y좌표 모두 (-100, +100) 사이의 int 값으로 생성

//--------
int main()
//--------
{
	Point2D points[100];

	for (int i = 0; i < 100; ++i) {
		cout << "(" << points[i].x << ", " << points[i].y << ")" << endl;
	}

	// (힌트) 제곱근을 계산해 주는 함수 - sqrt
	cout << "범위 안쪽의 점들" << endl;
	for (int i = 0; i < 100; ++i) {
		if ( sqrt( ( points[i].x * points[i].x ) + ( points[i].y * points[i].y )) <= 20 ) {
			cout << "(" << points[i].x << ", " << points[i].y << ")" << endl;
		}
	}

	//save("main.cpp");
}

//----------------------------------------------------------------------------------------------------------
#include <iostream>
#include <string>
#include "save.h"
using namespace std;

class STRING {
public :
	STRING(const char* str) {
		len = strlen(str);
		p = new char[len];
		// 복사는 반드시 이렇게 - DMA(Direct Memory Access)
		memcpy(p, str, len);
		// 관찰메시지
	}

	unsigned length() {
		return len;
	}

	void show() {
		for (int i = 0; i < len; ++i)
			cout << p[i];
		cout << endl;
	}

private :
	unsigned len{};
	char* p{};
};

//--------

//--------
{
	STRING s{ "2025년 11월 3일 9주 1일 - 지금부터 코드를 잘 따라 작성해 주세요" };

public:
	INT() = default;
	INT( int n ) : num{ n } {}

	INT(const INT&) = default;				// 자동으로 만들어진다
	INT& operator=(const INT&) = default;	// 자동으로 만들어진다

	// pre-increment
	INT& operator++() {
		++num;
		return *this;
	}

	// post-increment
	INT operator++( int ) {
		INT old = *this;	// 내 현재 값을 저장 -> 복사객체 old를 만들 수 밖에 없다
		++(*this);			// 나를 1증가하는 코드는 이미 만든 pre-increment를 이용해야.
		return old;			// 저장한 증가전의 나를 리턴
	}

private:
	int num;

	friend ostream& operator<<(ostream& os, const INT& rhs) {
		os << rhs.num;
		return os;
	}
};

int 오름(const void* a, const void* b)
{

}

//--------
{
	STRING s[] { "333", "1", "55555", "4444", "22" };

	관찰 = true;
	
	std::sort(begin(s), end(s), []( STRING& a, STRING& b) {
		return a.length() - b.length();
		});

	// return a.length() < b.length(); 로 바꾸면 동작한다.

	관찰 = false;

	for (const STRING& str : s)
		cout << str << endl;

	//save("main.cpp");

// [문제] main이 의도대로 실행되도록 class Dog를 coding 하라.
// class Dog는
// class Dog {
// private:
//    int		id;		// 생성시 [1, 100000] 사이의 랜덤값으로 설정
//    STRING	name;	// 생성시 [10, 50] 랜덤길이로 설정하고, 글자는 임의의 소문자로 설정
// };

std::default_random_engine dre;
std::uniform_int_distribution uidId{ 1, 10'0000 };
std::uniform_int_distribution uidLen{ 10, 50 };
std::uniform_int_distribution<int> uidChar{ 'a', 'z' };

class Dog {
public:
	Dog() {
		id = uidId(dre);
		// 길이와 이름
		int len = uidLen(dre);
		char* p = new char[len + 1];
		for (int i = 0; i < len; ++i)
			p[i] = uidChar(dre);
		p[len] = '\0';

		STRING temp(p);
		delete[] p;

		name = temp;		// name = move(temp); 도 상관없이 돌아간다
	}
private:
	int id;
	STRING name;

	friend ostream& operator<<(ostream& os, const Dog& dog) {
		print("[{:6}] - ", dog.id);
		cout << dog.name;
		return os;
	}
};

//--------
int main()
//--------
{
	관찰 = true;
	Dog dogs[10];

	for (const Dog& dog : dogs)
		cout << dog << endl;		// 화면에 [id는 6칸] - name
	
	//save("main.cpp");
}

// [문제] 다운받은 파일 "개들.txt" 에는 1000개의 Dog 객체가 있다.
// class Dog의 멤버는 다음과 같다.
//
// class Dog {
// private:
//		int				id;
//		std::string		name;
// };
//
// 파일에 있는 Dog 객체를 메모리에 저장하라.
// id 기준 오름차순으로 정렬하라.
// 화면에 id와 name을 출력하라.

//--------
int main()
//--------
{
	ifstream in{ "개들.txt" };
	if (not in)
		return 20251124;

	int id;
	string name;

	int cnt{};
	while (in >> id >> name) {
		cout << id << " ----- " << name << endl;
		++cnt;
	}
	cout << "모두" << cnt << "개의 Dog를 읽었다" << endl;

	//(*save)("main.cpp");
}
